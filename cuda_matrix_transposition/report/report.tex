\begin{algorithm}
    \caption{Naive Matrix Transposition}
    \begin{algorithmic}[1]
        \Procedure{NaiveTranspose}{$SRC, DST$}
            \For{$i = 0$ to $N$}
                \For{$j = 0$ to $N$}
                    \State $B[i][j] = A[j][i]$
                \EndFor
            \EndFor
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
\begin{algorithm}
    \caption{Matrix Transpose with Shared Memory}
    \begin{algorithmic}[1]
        \Procedure{TransposeSharedMem}{$SRC, DST$}
            \For{$i = 0$ to $N$}
                \For{$j = 0$ to $N$}
                    \State $B[i][j] = A[j][i]$
                \EndFor
            \EndFor
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
